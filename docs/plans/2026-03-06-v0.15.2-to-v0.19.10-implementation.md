```typescript
// src/assets/image/iccProfile.ts

/**
 * ICC color profile extraction and embedding for image optimization.
 *
 * Handles sRGB, Adobe RGB, ProPhoto RGB, and arbitrary ICC profiles.
 * Preserves profiles during recompression to prevent color shifts
 * in print workflows.
 */

import type { PdfStream, PdfRef, PdfDict, PdfArray, PdfName, PdfObjectRegistry } from '../../core/pdfObjects.js';

/** Parsed ICC profile metadata. */
export interface IccProfile {
  /** Raw ICC profile bytes. */
  readonly data: Uint8Array;
  /** Number of color components (1=gray, 3=RGB, 4=CMYK). */
  readonly components: number;
  /** Profile color space signature (e.g., 'RGB ', 'CMYK', 'GRAY'). */
  readonly colorSpace: string;
  /** Profile description from 'desc' tag, if present. */
  readonly description?: string;
}

/** Well-known profile signatures for fast comparison. */
const SRGB_PROFILE_SIZE_RANGE = [3000, 4000] as const;

/**
 * Extract ICC profile from a PDF image XObject's /ColorSpace entry.
 *
 * Handles:
 * - [/ICCBased <stream ref>] — extracts profile bytes from the stream
 * - [/Indexed [/ICCBased <stream ref>] ...] — extracts from base color space
 *
 * @returns The ICC profile data, or undefined if no ICC profile is present.
 */
export function extractIccProfile(
  colorSpaceEntry: PdfArray | PdfName | undefined,
  registry: PdfObjectRegistry,
): IccProfile | undefined {
  if (!colorSpaceEntry || colorSpaceEntry.kind !== 'array') return undefined;

  const arr = colorSpaceEntry as PdfArray;
  const first = arr.items[0];
  if (!first || first.kind !== 'name') return undefined;

  const csName = (first as PdfName).value.replace(/^\//, '');

  if (csName === 'ICCBased') {
    return extractFromIccBasedStream(arr, registry);
  }

  if (csName === 'Indexed') {
    // Base color space is items[1]
    const base = arr.items[1];
    if (base && base.kind === 'array') {
      return extractIccProfile(base as PdfArray, registry);
    }
  }

  return undefined;
}

function extractFromIccBasedStream(
  arr: PdfArray,
  registry: PdfObjectRegistry,
): IccProfile | undefined {
  const profileRef = arr.items[1];
  if (!profileRef || profileRef.kind !== 'ref') return undefined;

  const profile = registry.resolve(profileRef as PdfRef);
  if (!profile || profile.kind !== 'stream') return undefined;

  const stream = profile as PdfStream;
  const nEntry = stream.dict.get('/N');
  const components = nEntry && nEntry.kind === 'number'
    ? (nEntry as any).value as number
    : 3;

  const data = stream.getDecodedData?.() ?? stream.data;
  const colorSpace = parseIccColorSpace(data);
  const description = parseIccDescription(data);

  return { data, components, colorSpace, description };
}

/**
 * Read the color space signature from ICC profile header (bytes 16-19).
 * Per ICC spec, the 4-byte signature at offset 16 identifies the color space.
 */
function parseIccColorSpace(data: Uint8Array): string {
  if (data.length < 20) return 'RGB ';
  return String.fromCharCode(data[16], data[17], data[18], data[19]);
}

/**
 * Read the profile description from the 'desc' tag.
 * Simple extraction — reads ASCII description from the first desc tag found.
 */
function parseIccDescription(data: Uint8Array): string | undefined {
  if (data.length < 132) return undefined;

  // Tag table starts at offset 128
  const tagCount = (data[128] << 24) | (data[129] << 16) | (data[130] << 8) | data[131];
  if (tagCount < 1 || tagCount > 100) return undefined;

  for (let i = 0; i < tagCount; i++) {
    const base = 132 + i * 12;
    if (base + 12 > data.length) break;

    const sig = String.fromCharCode(data[base], data[base + 1], data[base + 2], data[base + 3]);
    if (sig !== 'desc') continue;

    const offset = (data[base + 4] << 24) | (data[base + 5] << 16) | (data[base + 6] << 8) | data[base + 7];
    const size = (data[base + 8] << 24) | (data[base + 9] << 16) | (data[base + 10] << 8) | data[base + 11];

    if (offset + size > data.length || size < 12) break;

    // 'desc' type: 4 bytes sig + 4 bytes reserved + 4 bytes count + ASCII
    const strLen = (data[offset + 8] << 24) | (data[offset + 9] << 16) | (data[offset + 10] << 8) | data[offset + 11];
    if (strLen < 1 || strLen > 256) break;

    const start = offset + 12;
    const end = Math.min(start + strLen - 1, data.length); // -1 for null terminator
    return new TextDecoder().decode(data.subarray(start, end));
  }

  return undefined;
}

/**
 * Build a PDF /ColorSpace array entry that references an ICC profile.
 *
 * Creates: [/ICCBased <stream ref>]
 * The stream contains the raw ICC profile bytes with /N set to the component count.
 */
export function buildIccColorSpace(
  profile: IccProfile,
  registry: PdfObjectRegistry,
): PdfArray {
  const { PdfName: PN, PdfNumber: PNum, PdfStream: PS, PdfArray: PA, PdfDict: PD } =
    await import('../../core/pdfObjects.js');

  const profileDict = new PD(new Map([
    ['/N', new PNum(profile.components)],
    ['/Filter', new PN('/FlateDecode')],
  ]));

  const profileStream = new PS(profileDict, profile.data);
  const ref = registry.register(profileStream);

  return new PA([new PN('/ICCBased'), ref]);
}
```

```typescript
import { describe, it, expect } from 'vitest';
import { downscaleImage } from '../../src/assets/image/imageOptimize.js';
import type { RawImageData } from '../../src/assets/image/imageOptimize.js';

describe('Lanczos3 resampling', () => {
  function makeCheckerboard(w: number, h: number): RawImageData {
    const pixels = new Uint8Array(w * h * 3);
    for (let y = 0; y < h; y++) {
      for (let x = 0; x < w; x++) {
        const i = (y * w + x) * 3;
        const v = ((x + y) & 1) ? 255 : 0;
        pixels[i] = pixels[i + 1] = pixels[i + 2] = v;
      }
    }
    return { pixels, width: w, height: h, channels: 3, bitsPerChannel: 8 };
  }

  it('preserves sharp edges better than bilinear', () => {
    const img = makeCheckerboard(100, 100);
    const lanczos = downscaleImage(img, { maxWidth: 50, maxHeight: 50, algorithm: 'lanczos' });
    const bilinear = downscaleImage(img, { maxWidth: 50, maxHeight: 50, algorithm: 'bilinear' });

    // Lanczos should have higher contrast (more pure black/white pixels)
    const lanczosContrast = computeContrast(lanczos.pixels);
    const bilinearContrast = computeContrast(bilinear.pixels);
    expect(lanczosContrast).toBeGreaterThan(bilinearContrast);
  });

  it('handles single-pixel images', () => {
    const img: RawImageData = {
      pixels: new Uint8Array([128, 64, 32]),
      width: 1, height: 1, channels: 3, bitsPerChannel: 8,
    };
    const result = downscaleImage(img, { maxWidth: 1, maxHeight: 1, algorithm: 'lanczos' });
    expect(result.width).toBe(1);
    expect(result.height).toBe(1);
  });

  it('preserves dimensions when target >= source', () => {
    const img = makeCheckerboard(50, 50);
    const result = downscaleImage(img, { maxWidth: 100, maxHeight: 100, algorithm: 'lanczos' });
    expect(result.width).toBe(50);
    expect(result.height).toBe(50);
  });
});

function computeContrast(pixels: Uint8Array): number {
  let extremes = 0;
  for (let i = 0; i < pixels.length; i += 3) {
    const v = pixels[i];
    if (v < 32 || v > 224) extremes++;
  }
  return extremes / (pixels.length / 3);
}
```

```typescript
// src/assets/image/compressionAnalysis.ts

import type { PdfDocument } from '../../core/pdfDocument.js';
import { extractImages, decodeImageStream } from './imageExtract.js';
import type { ImageInfo } from './imageExtract.js';
import { encodeJpegWasm, isJpegWasmReady } from '../../wasm/jpeg/bridge.js';
import { isGrayscaleImage } from './grayscaleDetect.js';
import { computeImageDpi } from './dpiAnalyze.js';

/** Per-image analysis result showing current vs optimal encoding. */
export interface ImageAnalysis {
  readonly name: string;
  readonly pageIndex: number;
  readonly width: number;
  readonly height: number;
  readonly currentSize: number;
  readonly currentFormat: string;
  readonly colorSpace: string;
  readonly estimatedJpegSize: number;
  readonly estimatedSavings: number;
  readonly savingsPercent: number;
  readonly isGrayscale: boolean;
  readonly effectiveDpi: number | undefined;
  readonly recommendation: 'recompress' | 'keep' | 'downscale' | 'grayscale';
}

/** Full document analysis report. */
export interface AnalysisReport {
  readonly images: readonly ImageAnalysis[];
  readonly totalCurrentSize: number;
  readonly totalEstimatedSize: number;
  readonly totalSavings: number;
  readonly totalSavingsPercent: number;
}

/**
 * Analyze all images in a PDF document and report potential savings
 * without modifying the document.
 */
export function analyzeImages(
  doc: PdfDocument,
  options: {
    readonly quality?: number;
    readonly maxDpi?: number;
  } = {},
): AnalysisReport {
  const quality = options.quality ?? 80;
  const images = extractImages(doc);
  const analyses: ImageAnalysis[] = [];
  let totalCurrent = 0;
  let totalEstimated = 0;

  for (const img of images) {
    const analysis = analyzeImage(img, quality, options.maxDpi);
    analyses.push(analysis);
    totalCurrent += analysis.currentSize;
    totalEstimated += analysis.estimatedJpegSize;
  }

  const totalSavings = totalCurrent - totalEstimated;

  return {
    images: analyses,
    totalCurrentSize: totalCurrent,
    totalEstimatedSize: totalEstimated,
    totalSavings: Math.max(0, totalSavings),
    totalSavingsPercent: totalCurrent > 0
      ? Math.round((Math.max(0, totalSavings) / totalCurrent) * 100)
      : 0,
  };
}

function analyzeImage(
  img: ImageInfo,
  quality: number,
  maxDpi?: number,
): ImageAnalysis {
  let estimatedSize = img.compressedSize;
  let recommendation: ImageAnalysis['recommendation'] = 'keep';
  let grayscale = false;

  // Try to decode and estimate JPEG size
  if (isJpegWasmReady() && img.bitsPerComponent === 8 && img.channels >= 1 && img.channels <= 4) {
    try {
      const decoded = decodeImageStream(img);
      if (decoded.length > 0) {
        grayscale = img.channels >= 3 && isGrayscaleImage(decoded, img.width, img.height, img.channels as 3 | 4);

        const channels = grayscale ? 1 : Math.min(img.channels, 3) as 1 | 3;
        const encoded = encodeJpegWasm(decoded, img.width, img.height, channels, quality);
        if (encoded) {
          estimatedSize = encoded.length;
          const savings = img.compressedSize - estimatedSize;
          if (savings > img.compressedSize * 0.1) {
            recommendation = grayscale ? 'grayscale' : 'recompress';
          }
        }
      }
    } catch {
      // Can't decode — keep original
    }
  }

  const dpi = maxDpi ? computeImageDpi(img.width, img.height, img.width * 72 / 150, img.height * 72 / 150) : undefined;
  if (dpi && dpi.horizontal > (maxDpi ?? 300)) {
    recommendation = 'downscale';
  }

  return {
    name: img.name,
    pageIndex: img.pageIndex,
    width: img.width,
    height: img.height,
    currentSize: img.compressedSize,
    currentFormat: img.filters.join('+') || 'raw',
    colorSpace: img.colorSpace,
    estimatedJpegSize: estimatedSize,
    estimatedSavings: Math.max(0, img.compressedSize - estimatedSize),
    savingsPercent: img.compressedSize > 0
      ? Math.round((Math.max(0, img.compressedSize - estimatedSize) / img.compressedSize) * 100)
      : 0,
    isGrayscale: grayscale,
    effectiveDpi: dpi?.horizontal,
    recommendation,
  };
}
```

```typescript
// src/assets/image/jpegMarkers.ts

/** JPEG Start of Frame marker types. */
export const enum JpegSofType {
  BASELINE = 0xC0,          // SOF0 — Baseline DCT (Huffman)
  EXTENDED = 0xC1,          // SOF1 — Extended sequential DCT (Huffman)
  PROGRESSIVE = 0xC2,       // SOF2 — Progressive DCT (Huffman)
  LOSSLESS = 0xC3,          // SOF3 — Lossless (Huffman)
  ARITHMETIC_SEQUENTIAL = 0xC9,  // SOF9 — Extended sequential DCT (Arithmetic)
  ARITHMETIC_PROGRESSIVE = 0xCA, // SOF10 — Progressive DCT (Arithmetic)
  ARITHMETIC_LOSSLESS = 0xCB,    // SOF11 — Lossless (Arithmetic)
}

/** Result of JPEG marker analysis. */
export interface JpegMarkerInfo {
  readonly isArithmeticCoded: boolean;
  readonly isProgressive: boolean;
  readonly sofType: number;
  readonly width: number;
  readonly height: number;
  readonly components: number;
  readonly bitsPerComponent: number;
}

/**
 * Parse JPEG markers to detect arithmetic coding and other properties.
 * Scans marker segments without decoding image data.
 */
export function analyzeJpegMarkers(data: Uint8Array): JpegMarkerInfo | undefined {
  if (data.length < 4 || data[0] !== 0xFF || data[1] !== 0xD8) return undefined;

  let offset = 2;
  while (offset < data.length - 1) {
    if (data[offset] !== 0xFF) { offset++; continue; }

    // Skip padding 0xFF bytes
    while (offset < data.length && data[offset] === 0xFF) offset++;
    if (offset >= data.length) break;

    const marker = data[offset++];

    // SOF markers (0xC0-0xCF except 0xC4=DHT, 0xC8=JPG, 0xCC=DAC)
    if (marker >= 0xC0 && marker <= 0xCF && marker !== 0xC4 && marker !== 0xC8 && marker !== 0xCC) {
      if (offset + 7 > data.length) break;
      const len = (data[offset] << 8) | data[offset + 1];
      const bpc = data[offset + 2];
      const height = (data[offset + 3] << 8) | data[offset + 4];
      const width = (data[offset + 5] << 8) | data[offset + 6];
      const components = data[offset + 7];

      return {
        isArithmeticCoded: marker >= 0xC9 && marker <= 0xCB,
        isProgressive: marker === 0xC2 || marker === 0xCA,
        sofType: marker,
        width,
        height,
        components,
        bitsPerComponent: bpc,
      };
    }

    // Skip over marker segment
    if (marker === 0xD9 || marker === 0x00) continue; // EOI or stuffed byte
    if (marker >= 0xD0 && marker <= 0xD7) continue; // RST markers (no length)
    if (offset + 1 >= data.length) break;

    const segLen = (data[offset] << 8) | data[offset + 1];
    offset += segLen;
  }

  return undefined;
}
```

```typescript
async function processWithConcurrency<T>(
  items: readonly T[],
  concurrency: number,
  processor: (item: T, index: number) => Promise<void>,
): Promise<void> {
  let nextIndex = 0;
  const workers = Array.from({ length: Math.min(concurrency, items.length) }, async () => {
    while (nextIndex < items.length) {
      const index = nextIndex++;
      await processor(items[index], index);
    }
  });
  await Promise.all(workers);
}
```

```typescript
// src/browser/worker.ts

/**
 * Web Worker wrapper for offloading PDF generation off the main thread.
 */

export interface PdfWorkerOptions {
  /** URL to the worker script. If not provided, uses inline blob worker. */
  readonly workerUrl?: string | URL;
}

export class PdfWorker {
  private worker: Worker;
  private nextId = 0;
  private pending = new Map<number, { resolve: (v: Uint8Array) => void; reject: (e: Error) => void }>();

  constructor(options: PdfWorkerOptions = {}) {
    if (options.workerUrl) {
      this.worker = new Worker(options.workerUrl, { type: 'module' });
    } else {
      const blob = new Blob([WORKER_SCRIPT], { type: 'application/javascript' });
      this.worker = new Worker(URL.createObjectURL(blob), { type: 'module' });
    }

    this.worker.onmessage = (e: MessageEvent) => {
      const { id, result, error } = e.data;
      const handler = this.pending.get(id);
      if (!handler) return;
      this.pending.delete(id);
      if (error) handler.reject(new Error(error));
      else handler.resolve(result);
    };
  }

  /**
   * Execute a PDF generation function in the worker thread.
   * The function receives the library as its argument and must return Uint8Array.
   */
  async execute(taskFn: string): Promise<Uint8Array> {
    const id = this.nextId++;
    return new Promise((resolve, reject) => {
      this.pending.set(id, { resolve, reject });
      this.worker.postMessage({ id, task: taskFn });
    });
  }

  terminate(): void {
    this.worker.terminate();
    for (const { reject } of this.pending.values()) {
      reject(new Error('Worker terminated'));
    }
    this.pending.clear();
  }
}

const WORKER_SCRIPT = `
import * as pdf from 'modern-pdf-lib';
self.onmessage = async (e) => {
  const { id, task } = e.data;
  try {
    const fn = new Function('pdf', task);
    const result = await fn(pdf);
    self.postMessage({ id, result }, [result.buffer]);
  } catch (err) {
    self.postMessage({ id, error: err.message });
  }
};
`;
```

```typescript
export function saveAsDownload(bytes: Uint8Array, filename: string = 'document.pdf'): void {
  const blob = new Blob([bytes], { type: 'application/pdf' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
```

```typescript
/**
 * Generate PDF operators to draw a QR code at the specified position.
 */
export function qrCodeOperators(
  matrix: QrCodeMatrix,
  x: number,
  y: number,
  options?: QrCodeOptions,
): string {
  const moduleSize = options?.moduleSize ?? 2;
  const quietZone = options?.quietZone ?? 4;
  const totalSize = (matrix.size + quietZone * 2) * moduleSize;

  let ops = '';
  ops += saveState();

  // Background
  if (options?.backgroundColor) {
    ops += applyFillColor(options.backgroundColor);
    ops += rectangle(x, y, totalSize, totalSize);
    ops += fill();
  }

  // Foreground modules
  ops += applyFillColor(options?.color ?? { type: 'grayscale', gray: 0 });
  for (let row = 0; row < matrix.size; row++) {
    for (let col = 0; col < matrix.size; col++) {
      if (matrix.modules[row * matrix.size + col]) {
        const mx = x + (col + quietZone) * moduleSize;
        const my = y + totalSize - (row + quietZone + 1) * moduleSize;
        ops += rectangle(mx, my, moduleSize, moduleSize);
      }
    }
  }
  ops += fill();
  ops += restoreState();

  return ops;
}
```

```typescript
/** Full table configuration. */
export interface DrawTableOptions {
  readonly x: number;
  readonly y: number;
  readonly width: number;
  readonly rows: readonly TableRow[];
  readonly columns?: readonly TableColumn[];
  readonly font: FontRef;
  readonly fontSize?: number;
  readonly textColor?: Color;
  readonly borderColor?: Color;
  readonly borderWidth?: number;
  readonly headerRows?: number;
  readonly padding?: number;
}

/** Result of table rendering. */
export interface TableRenderResult {
  readonly width: number;
  readonly height: number;
  readonly rowHeights: readonly number[];
  readonly columnWidths: readonly number[];
  readonly pagesUsed: number;
}

/**
 * Render a table to PDF operators.
 */
export function renderTable(options: DrawTableOptions): { ops: string; result: TableRenderResult } {
  const fontSize = options.fontSize ?? 12;
  const padding = options.padding ?? 4;
  const borderWidth = options.borderWidth ?? 0.5;
  const borderColor = options.borderColor ?? { type: 'grayscale' as const, gray: 0 };
  const textColor = options.textColor ?? { type: 'grayscale' as const, gray: 0 };

  // Calculate column widths
  const colWidths = resolveColumnWidths(options);
  const rowHeights = resolveRowHeights(options, colWidths, fontSize, padding);

  let ops = '';
  ops += saveState();

  let currentY = options.y;

  for (let rowIdx = 0; rowIdx < options.rows.length; rowIdx++) {
    const row = options.rows[rowIdx];
    const rowHeight = rowHeights[rowIdx];
    let currentX = options.x;

    // Row background
    if (row.backgroundColor) {
      ops += applyFillColor(row.backgroundColor);
      ops += rectangle(currentX, currentY - rowHeight, options.width, rowHeight);
      ops += fill();
    }

    // Cell content and borders
    for (let colIdx = 0; colIdx < colWidths.length; colIdx++) {
      const cell = normalizeCell(row.cells[colIdx]);
      const cellWidth = colWidths[colIdx];

      // Cell background
      if (cell.backgroundColor) {
        ops += applyFillColor(cell.backgroundColor);
        ops += rectangle(currentX, currentY - rowHeight, cellWidth, rowHeight);
        ops += fill();
      }

      // Cell border
      ops += applyStrokeColor(borderColor);
      ops += setLineWidth(borderWidth);
      ops += rectangle(currentX, currentY - rowHeight, cellWidth, rowHeight);
      ops += stroke();

      // Cell text
      const textX = currentX + padding;
      const textY = currentY - padding - fontSize;
      ops += applyFillColor(cell.textColor ?? textColor);
      ops += beginText();
      ops += setFont(options.font.name, cell.fontSize ?? fontSize);
      ops += moveText(textX, textY);
      ops += showText(cell.content);
      ops += endText();

      currentX += cellWidth;
    }

    currentY -= rowHeight;
  }

  ops += restoreState();

  return {
    ops,
    result: {
      width: options.width,
      height: options.y - currentY,
      rowHeights,
      columnWidths: colWidths,
      pagesUsed: 1,
    },
  };
}
```
